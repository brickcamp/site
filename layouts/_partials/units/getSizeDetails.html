{{/* Convert size params into short-/longtext */}} {{- $units := site.Data.entries.units.lengths -}} {{- $size := . -}} {{- $result := dict -}} {{- $short := slice -}} {{- $long := slice -}} {{- range $size -}} {{- $value := . | replaceRE `[^0-9]` "" -}} {{- $key := . | replaceRE `[0-9]` "" -}} {{- $unit := index $units $key -}} {{- $label := cond (eq $value "1") $unit.one $unit.many -}} {{- $short = $short | append (print $value $key) -}} {{- $long = $long | append (printf "%s %s" $value $label) -}} {{- end -}} {{- $short = delimit $short " × " -}} {{- $long = delimit $long " × " -}} {{- return dict "short" $short "long" $long -}}
